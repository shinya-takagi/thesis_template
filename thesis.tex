      \huge\headfont\@chapapp\thechapter\@chappos\quad%
      %% \par\nobreak
      %% \vskip \Cvs % English 20pt
    \fi
    \interlinepenalty\@M%
    \Huge \headfont{} #1\par\nobreak%
    \vskip 3\Cvs}} % English 40pt
